getPublicDigest(pk)

Constructs public key/subkey digest

pk key/subkey packet serializable object

Returns the message digest as string

function getPublicDigest(pk) { var n = 6; var npkey = misc.pubkey_get_npkey(pk.pubkey_algo); if (pk.version < 4) n+=2; var pkey_digest = ""; for (var i=0;i<npkey;i++) { pkey_digest += pk.pkey[i]; n += pk.pkey[i].length; } return String.fromCharCode(0x99) + String.fromCharCode(n >> 8) + String.fromCharCode(n & 0xff) + String.fromCharCode(pk.version) + misc.u32_to_string(pk.timestamp) + String.fromCharCode(pk.pubkey_algo) + pkey_digest; }
