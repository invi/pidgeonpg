Key.prototype.addPacket = function(pkt) { try { switch (pkt.pkttype) { case PGP.PKT.PUBLIC_SUBKEY: case PGP.PKT.SECRET_SUBKEY: var subkey = Subkey.load({pkt: pkt}, this) this.subkeys.push(subkey); this._key.subkeys.push(subkey._subkey); return subkey; case PGP.PKT.USER_ID: var uid = UserId.load({pkt: pkt}, this); this.uids.push(uid); this._key.uids.push(uid._uid); return uid; case PGP.PKT.SIGNATURE: switch(pkt.sig_class) { case PGP.SIGCLASS.SUBKEY_SIG: case PGP.SIGCLASS.SUBKEY_REV: var subkey = this.subkeys[this.subkeys.length-1]; return subkey.addPacket(pkt); case PGP.SIGCLASS.UID_SIG: case PGP.SIGCLASS.CASUAL_SIG: case PGP.SIGCLASS.UID_REV: case PGP.SIGCLASS.KEY_SIG: var uid = this.uids[this.uids.length-1]; return uid.addPacket(pkt); case PGP.SIGCLASS.KEY_REV: this._key.revsigs.push(pkt); var sig = Signature.load({pkt: pkt}, this); this.revsigs.push(sig); return sig; default: throw new Error("Invalid Key signature class: " + pkt.sig_class); } case PGP.PKT.RING_TRUST: throw new Error("Ring trust packet not implemented"); default: throw new Error("Invalid Key packet type: " + pkt.pkttype); } } catch(err) {console.trace() ; logger.error(err);} }
